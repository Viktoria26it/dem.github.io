import pandas as pd import numpy as np from sklearn.model_selection import train_test_split from sklearn.tree import DecisionTreeClassifier, plot_tree import matplotlib.pyplot as plt import joblib # ========================================== # 1. ГЕНЕРАЦИЯ ДАТАСЕТА КАСКО # ========================================== np.random.seed(42) n_samples = 1000 # Возраст: от 18 до 75 лет age = np.random.randint(18, 76, n_samples) # Стаж вождения: от 0 до (возраст - 18) лет driving_experience = [np.random.randint(0, max(1, a - 18)) for a in age] # Число ДТП: зависит от стажа и возраста accidents = [] for i in range(n_samples): risk_factor = 0 if age[i] < 23: risk_factor += 2 if driving_experience[i] < 3: risk_factor += 2 if age[i] > 50 and driving_experience[i] > 20: risk_factor -= 1 prob = max(0.05, 0.1 + (risk_factor * 0.1)) acc_count = np.random.poisson(prob * 5) accidents.append(acc_count) # Пол (для демонстрации кодирования, как в вашем примере) gender = np.random.choice(['M', 'F'], n_samples) # Целевая переменная: Решение по КАСКО (1 - Одобрить, 0 - Отказать) insurance_approved = [] for i in range(n_samples): if accidents[i] >= 3: insurance_approved.append(0) elif age[i] < 21 and driving_experience[i] < 1 and accidents[i] > 0: insurance_approved.append(0) else: insurance_approved.append(1) # Создание DataFrame df = pd.DataFrame({ 'age': age, 'driving_experience': driving_experience, 'accidents_count': accidents, 'gender': gender, 'insurance_approved': insurance_approved }) print("=" * 50) print("ПРИМЕР ДАТАСЕТА КАСКО") print("=" * 50) print(df.head(10)) print(f"\nРаспределение решений: {df['insurance_approved'].value_counts().to_dict()}") # ========================================== # 2. ПРЕОБРАЗОВАНИЕ НЕЧИСЛОВЫХ АТРИБУТОВ # ========================================== print("\n" + "=" * 50) print("КОДИРОВАНИЕ КАТЕГОРИАЛЬНЫХ ПРИЗНАКОВ") print("=" * 50) print("\nДо кодирования:") print(df[['gender']].head()) # One-Hot Encoding (как в вашем примере) df_encoded = pd.get_dummies(df, columns=['gender'], drop_first=True) print("\nПосле кодирования:") print(df_encoded.head()) print(f"\nКолонки после кодирования: {df_encoded.columns.tolist()}") # ========================================== # 3. РАЗБИВКА НА ПРИЗНАКИ И ЦЕЛЕВУЮ ПЕРЕМЕННУЮ # ========================================== x = df_encoded.drop('insurance_approved', axis=1) y = df_encoded['insurance_approved'] print("\n" + "=" * 50) print("ВХОДНЫЕ ПРИЗНАКИ (X) и ЦЕЛЬ (y)") print("=" * 50) print(f"Признаки: {x.columns.tolist()}") print(f"Целевая переменная: insurance_approved") print(f"Размер X: {x.shape}, Размер y: {y.shape}") # ========================================== # 4. РАЗБИВКА НА ТРЕНИРОВОЧНУЮ И ТЕСТОВУЮ ЧАСТЬ # ========================================== X_train, X_test, y_train, y_test = train_test_split( x, # Признаки (возраст, стаж, ДТП, пол) y, # Цель (одобрение КАСКО) test_size=0.2, # 20% данных уйдет в тест, 80% - в обучение random_state=42 # Фиксирует случайность для повторяемости результата ) print("\n" + "=" * 50) print("РАЗБИВКА ДАННЫХ") print("=" * 50) print(f"Обучающая выборка: {X_train.shape[0]} записей") print(f"Тестовая выборка: {X_test.shape[0]} записей") # ========================================== # 5. СОЗДАНИЕ ДЕРЕВА РЕШЕНИЙ # ========================================== tree = DecisionTreeClassifier(max_depth=5, random_state=42) print("\n" + "=" * 50) print("ОБУЧЕНИЕ МОДЕЛИ") print("=" * 50) # ========================================== # 6. ОБУЧЕНИЕ ДЕРЕВА НА ТРЕНИРОВОЧНЫХ ДАННЫХ # ========================================== tree.fit(X_train, y_train) print("Модель успешно обучена!") # ========================================== # 7. ПРОВЕРКА РАБОТЫ ДЕРЕВА НА ТЕСТОВОЙ ЧАСТИ # ========================================== train_score = tree.score(X_train, y_train) test_score = tree.score(X_test, y_test) print("\n" + "=" * 50) print("ТОЧНОСТЬ МОДЕЛИ") print("=" * 50) print(f"Точность на обучающих данных: {train_score:.4f} ({train_score*100:.2f}%)") print(f"Точность на тестовых данных: {test_score:.4f} ({test_score*100:.2f}%)") # ========================================== # 8. ВИЗУАЛИЗАЦИЯ ЛОГИКИ ПРИНЯТИЯ РЕШЕНИЙ # ========================================== plt.figure(figsize=(20, 10)) plot_tree( tree, feature_names=x.columns, class_names=['Отказ', 'Одобрено'], filled=True, rounded=True, fontsize=10 ) plt.title("Дерево решений - Страхование КАСКО") plt.savefig('casco_decision_tree.png', dpi=300, bbox_inches='tight') plt.show() print("\n" + "=" * 50) print("СОХРАНЕНИЕ РЕЗУЛЬТАТОВ") print("=" * 50) # Сохранение модели joblib.dump(tree, 'casco_tree_model.pkl') print("✓ Модель сохранена в 'casco_tree_model.pkl'") # Сохранение датасета df_encoded.to_csv('casco_dataset.csv', index=False) print("✓ Датасет сохранён в 'casco_dataset.csv'") # ========================================== # 9. ГЕНЕРАЦИЯ HTML СТРАНИЦЫ С ТОЧНОСТЬЮ # ========================================== html_content = f""" Мониторинг модели КАСКО

🛡️ Модель КАСКО

Система принятия решений по страхованию

{test_score:.2%}
Decision Tree Classifier
📊 Обучающая выборка: {train_score:.2%}
🧪 Тестовая выборка: {test_score:.2%}
📁 Записей в датасете: {n_samples}
🌳 Глубина дерева: 5
Дата обновления модели: 21 марта 2026 г.
""" with open('index.html', 'w', encoding='utf-8') as f: f.write(html_content) print("✓ HTML-страница создана в 'index.html'") print("\n" + "=" * 50) print("ВСЕ ФАЙЛЫ СОЗДАНЫ УСПЕШНО!") print("=" * 50) print("\nСписок файлов:") print(" 1. casco_dataset.csv - датасет для обучения") print(" 2. casco_tree_model.pkl - сохранённая модель") print(" 3. casco_decision_tree.png - визуализация дерева") print(" 4. index.html - страница с точностью для сайта") print("\nЗагрузите index.html на ваш хостинг для публикации!")
